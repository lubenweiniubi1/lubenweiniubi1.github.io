## WebSocket 详解与 Express + React 实战

### 一、WebSocket 是什么？

**WebSocket** 是一种在**单个 TCP 连接**上进行**全双工通信**的协议，允许服务器和客户端**双向实时通信**。

#### 核心特点：
- ✅ **全双工**：客户端和服务器可同时发送/接收数据
- ✅ **持久连接**：建立连接后保持长连接，无需反复握手
- ✅ **低延迟**：相比 HTTP 轮询，实时性大幅提升
- ✅ **轻量协议**：数据帧头部小，传输效率高

#### 与 HTTP 对比：
| 特性 | HTTP | WebSocket |
|------|------|-----------|
| 通信模式 | 请求-响应（单向） | 全双工（双向） |
| 连接 | 短连接（每次请求新建） | 长连接（一次握手持续通信） |
| 实时性 | 需轮询（延迟高） | 实时推送（延迟低） |
| 适用场景 | 传统网页、API调用 | 聊天、游戏、股票、协同编辑 |

#### 2. `server.js` 完整代码
```javascript
const express = require('express');
const http = require('http');
const WebSocket = require('ws');
const cors = require('cors');

const app = express();
const PORT = 3001;

// 允许跨域
app.use(cors());

// 静态文件服务（可选，用于测试）
app.use(express.static('public'));

// 创建 HTTP 服务器
const server = http.createServer(app);

// 创建 WebSocket 服务器
const wss = new WebSocket.Server({ server });

// 存储所有连接的客户端
const clients = new Set();

wss.on('connection', (ws) => {
  console.log('✅ 新客户端连接');

  // 添加到客户端集合
  clients.add(ws);

  // 发送欢迎消息
  ws.send(JSON.stringify({
    type: 'system',
    message: '欢迎连接到 WebSocket 服务器！'
  }));

  // 监听客户端消息
  ws.on('message', (data) => {
    try {
      const message = JSON.parse(data);
      console.log('📨 收到消息:', message);

      // 广播给所有客户端（包括发送者）
      clients.forEach(client => {
        if (client.readyState === WebSocket.OPEN) {
          client.send(JSON.stringify({
            type: 'chat',
            username: message.username || '匿名',
            message: message.text,
            timestamp: new Date().toISOString()
          }));
        }
      });
    } catch (error) {
      console.error('❌ 消息解析错误:', error);
    }
  });

  // 连接关闭
  ws.on('close', () => {
    console.log('❌ 客户端断开连接');
    clients.delete(ws);
  });

  // 错误处理
  ws.on('error', (error) => {
    console.error('❌ WebSocket 错误:', error);
  });
});

// 启动服务器
server.listen(PORT, () => {
  console.log(`🚀 服务器运行在 http://localhost:${PORT}`);
  console.log(`🔌 WebSocket 地址: ws://localhost:${PORT}`);
});
```

#### 2. `src/App.js` 完整代码
```jsx
import React, { useState, useEffect, useRef } from 'react';
import './App.css';

function App() {
  const [messages, setMessages] = useState([]);
  const [inputMessage, setInputMessage] = useState('');
  const [username, setUsername] = useState('用户' + Math.floor(Math.random() * 1000));
  const [isConnected, setIsConnected] = useState(false);
  const wsRef = useRef(null);
  const messagesEndRef = useRef(null);

  // 滚动到底部
  const scrollToBottom = () => {
    messagesEndRef.current?.scrollIntoView({ behavior: 'smooth' });
  };

  // 建立 WebSocket 连接
  useEffect(() => {
    const ws = new WebSocket('ws://localhost:3001');
    wsRef.current = ws;

    ws.onopen = () => {
      console.log('✅ WebSocket 连接成功');
      setIsConnected(true);
      addMessage('system', '已连接到服务器');
    };

    ws.onmessage = (event) => {
      try {
        const data = JSON.parse(event.data);
        console.log('📨 收到消息:', data);
        
        addMessage(data.type, data.message, data.username, data.timestamp);
      } catch (error) {
        console.error('❌ 消息解析错误:', error);
      }
    };

    ws.onclose = () => {
      console.log('❌ WebSocket 连接关闭');
      setIsConnected(false);
      addMessage('system', '与服务器断开连接');
    };

    ws.onerror = (error) => {
      console.error('❌ WebSocket 错误:', error);
      addMessage('system', '连接发生错误');
    };

    return () => {
      ws.close();
    };
  }, []);

  // 消息添加辅助函数
  const addMessage = (type, message, username = null, timestamp = null) => {
    setMessages(prev => [
      ...prev,
      {
        id: Date.now(),
        type,
        message,
        username,
        timestamp: timestamp || new Date().toISOString()
      }
    ]);
  };

  // 发送消息
  const sendMessage = (e) => {
    e.preventDefault();
    
    if (!inputMessage.trim() || !isConnected) return;

    const message = {
      username,
      text: inputMessage.trim()
    };

    wsRef.current.send(JSON.stringify(message));
    
    // 清空输入框
    setInputMessage('');
  };

  return (
    <div className="App">
      <header className="header">
        <h1>💬 WebSocket 聊天室</h1>
        <div className={`status ${isConnected ? 'connected' : 'disconnected'}`}>
          {isConnected ? '🟢 已连接' : '🔴 未连接'}
        </div>
      </header>

      <div className="chat-container">
        <div className="messages">
          {messages.map(msg => (
            <div 
              key={msg.id} 
              className={`message ${msg.type} ${msg.username === username ? 'self' : ''}`}
            >
              {msg.type === 'system' ? (
                <div className="system-message">
                  <span className="badge">📢 系统</span>
                  <span>{msg.message}</span>
                </div>
              ) : (
                <>
                  <div className="message-header">
                    <span className="username">{msg.username}</span>
                    <span className="timestamp">
                      {new Date(msg.timestamp).toLocaleTimeString('zh-CN')}
                    </span>
                  </div>
                  <div className="message-content">{msg.message}</div>
                </>
              )}
            </div>
          ))}
          <div ref={messagesEndRef} />
        </div>

        <form className="input-area" onSubmit={sendMessage}>
          <input
            type="text"
            value={inputMessage}
            onChange={(e) => setInputMessage(e.target.value)}
            placeholder="输入消息..."
            disabled={!isConnected}
          />
          <button type="submit" disabled={!isConnected || !inputMessage.trim()}>
            📤 发送
          </button>
        </form>
      </div>
    </div>
  );
}

export default App;
```

### 五、关键知识点总结

| 概念 | 说明 |
|------|------|
| `ws://` | WebSocket 协议前缀（非加密） |
| `wss://` | WebSocket Secure（加密，类似 HTTPS） |
| `onopen` | 连接建立时触发 |
| `onmessage` | 收到消息时触发 |
| `onclose` | 连接关闭时触发 |
| `onerror` | 发生错误时触发 |
| `send()` | 发送消息方法 |
| `JSON.stringify()` | WebSocket 只能发送字符串/二进制，对象需序列化 |
